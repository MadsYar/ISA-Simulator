0: LI R1, 1;			 # Used to loads value 1 from data memory
1: LD R2, R1;			 # Loads value 1 from data memory
2: LD R3, R4;			 # Loads value 0 from data memory
3: LI R10, 3; 			 # Value of 3 used for calculations
4: LI R5, 1; 			 # COUNTER
5: LI R9, 9;			 # Used for jumping
6: LI R11, 2; 			 # Defines which adress is used to save the original remaining value
7: LI R13, 14;			 # Used for jumping
8: LI R14, 17;			 # Used for jumping
9: SUB R2, R2, R3; 		 # SUBTRACTION
10: ADD R6, R6, R5; 		 # Storage of counter
11: JLT R9, R3, R2;		 # Loops if the number is still bigger than 15
12: JEQ R14, R2, R3; 		 # If the remaining number is fifteen then the program ends
13: SD R2, R11;  		 # Saving the remaining value that is non divisible in Address
14: SUB R2, R2, R10; 		 # Checks if the remaining non divisible ammount is divisible by 3 as it is the lowest common denominator of 15
15: ADD R7, R7, R5; 		 # New counter storage
16: JLT R13, R10, R2; 		 # loops if the number is still lower than 3
17: END
